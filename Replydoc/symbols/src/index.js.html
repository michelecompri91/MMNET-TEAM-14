 95 function checkForPendingIntent(){
 96 	var platformTools = require('bencoding.android.tools').createPlatform();
 97 	var isInForeground = platformTools.isInForeground();
 98 	if(isInForeground){
 99 		Ti.API.info("CHECK INTENT");
100 		var activity = Ti.Android.currentActivity;
101 		var intent = activity.intent;
102 		Ti.API.info(JSON.stringify(intent));
103 		
104 		if (intent.hasExtra('ntfId')) {
105 			Ti.API.info("*******found ntfID");
106 			// alert("found " + intent.getIntgExtra());
107 		}
108 		if (intent.hasExtra('payload')) { 
109 			Ti.API.info("*******found " + intent.getStringExtra('payload'));
110 			dbHelper.addQuestion(intent.getStringExtra('payload'));
111 		}
112 	}
113 }

115 function registerForNotification(){
116 	var gcm = require('net.iamyellow.gcmjs');
117 	var dbHelper = require('dbHelper');
118 	
119 	var pendingData = gcm.data;
120 	if (pendingData && pendingData !== null) {
121 		// if we're here is because user has clicked on the notification
122 		// and we set extras for the intent 
123 		// and the app WAS NOT running 
124 		Ti.API.info('******* data (started) ' + JSON.stringify(pendingData));
125 		// dbHelper.addQuestion(pendingData.payload);
126 	}
127 	
128 	gcm.registerForPushNotifications({
129 		success: function (ev) {  
130 			Alloy.Globals.gcm = ev.deviceToken;
131 			Ti.API.info('******* success, ' + ev.deviceToken);
132 		},
133 		error: function (ev) { 
134 			Ti.API.info('******* error, ' + ev.error);
135 		},
136 		callback: function (data) { 
137 			// alert(JSON.stringify(data));
138 			dbHelper.addQuestion(data.message);
139 			// Ti.API.info(data);
140 			// Ti.API.info(JSON.stringify(data.message));
141 		},
142 		unregister: function (ev) { 
143 			Ti.API.info('******* unregister, ' + ev.deviceToken);
144 		},
145 		data: function (data) {
146 			// if we're here is because user has clicked on the notification
147 			// and we set extras in the intent 
148 			// and the app WAS RUNNING (=> RESUMED) 
149 			Ti.API.info('******* data (resumed) ' + JSON.stringify(data));
150 			dbHelper.addQuestion(data.payload);
151 			// alert(data.payload);
152 		}
153 	});
154 	
155 	
156 	
157 }
