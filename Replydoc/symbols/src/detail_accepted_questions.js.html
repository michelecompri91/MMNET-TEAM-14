 61 function DownL_Image(){
 62 
 63 	var xhr = Titanium.Network.createHTTPClient({
 64 	onload: function() {
 65  // first, grab a "handle" to the file where you'll store the downloaded data
 66 		var f = Ti.Filesystem.getFile(Ti.Filesystem.applicationDataDirectory,'mygraphic.png');
 67 		f.write(this.responseData); // write to the file
 68 		Ti.App.fireEvent('graphic_downloaded', {filepath:f.nativePath});
 69 	},
 70 	timeout: 10000
 71 });
