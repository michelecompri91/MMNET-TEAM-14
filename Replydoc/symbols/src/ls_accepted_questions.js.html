 26 function insertIntoRow(collection){
 27 	var data = [];
 28 	for(var i=0; i<collection.length; i++){ 
 29 		Ti.API.info("lsAppcepted " + JSON.stringify(collection.at(i)));
 30 		var splittato = coll_question.at(i).get("content").split('&&&');
 31 		data.push({ 
 32 			"id_question": collection.at(i).get("id"),
 33 			"v_container":{"backgroundColor": (i%2 == 0) ? "#efefef": "white"},
 34 			"lbl_question":{"text": splittato[0] },//collection.at(i).get("content")},
 35 			"lbl_indirizzo":{"text": splittato[1] },
 36 			"lbl_date_time":{"text": formatDate(collection.at(i).get("timestamp"))}
 37 		});
 38 	}
 39 	return data;
 40 }
