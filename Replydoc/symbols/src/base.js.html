  3 openWindow = function(win, animate){ 
  4 	var last_win_id = Alloy.Globals.openWindows[Alloy.Globals.openWindows.length-1];
  5 	Ti.API.info(win.id + "   " + last_win_id);
  6 	if(last_win_id === win.id){
  7 		Ti.API.info(" ###########################################################");
  8 		Ti.API.info(" ######## This window is already open, nothing to do! ######");
  9 		Ti.API.info(" ###########################################################"); 
 10 	}
 11 	else{
 12 		win.open({animated:false});
 13 		Alloy.Globals.openWindows.push(win.id);
 14 	} 
 15 	
 16 	win.addEventListener('close', function(){
 17 		Ti.API.info('Got close event for window ' );
 18 		Alloy.Globals.openWindows.pop();
 19 	});
 20 };

 32 formatDate = function(timestamp){
 33 	var date = new Date(parseFloat(timestamp));
 34 	var monthDict = ["Jan", "Feb", "Mar", "Apr", "May", "Jun", "Jul", "Aug", "Sept", "Oct", "Nov", "Dec"];
 35 	return ( date.getDate() + " " + 
 36 			monthDict[date.getMonth()+1] + " " + 
 37 			date.getFullYear() + "   " + 
 38 			date.getHours() + ":" +
 39 			date.getMinutes() ) ;
 40 };
